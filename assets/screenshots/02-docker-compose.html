docker-compose.yml — Service Orchestration 
version: '3.8'

services:

  postgres:                          # 🗄️ PostgreSQL 16
    image: postgres:16-alpine
    ports: ["5432:5432"]
    environment:
      POSTGRES_DB:       shopsmart
      POSTGRES_USER:     postgres
      POSTGRES_PASSWORD: postgres
    volumes:
      - ./database/schema.sql → 01-schema.sql
      - ./database/seed.sql   → 02-seed.sql

  redis:                             # ⚡ Redis 7 (caching)
    image: redis:7-alpine
    ports: ["6379:6379"]

  api:                               # 🔷 TypeScript API (Fastify)
    build: ./api
    ports: ["3000:3000"]
    depends_on: [postgres, redis]
    environment:
      DB_HOST:     postgres
      REDIS_HOST:  redis
      JWT_SECRET:  super-secret-key…

  ml-service:                        # 🐍 Python ML Service (FastAPI)
    build: ./ml-service
    ports: ["8000:8000"]
    depends_on: [postgres]
    environment:
      POSTGRES_HOST: postgres
      POSTGRES_DB:   shopsmart
