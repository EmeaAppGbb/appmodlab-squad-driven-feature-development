 1-- ShopSmart Database Schema
 2
 3CREATE TABLE users (
 4    id          SERIAL PRIMARY KEY,
 5    email       VARCHAR(255) UNIQUE NOT NULL,
 6    password_hash VARCHAR(255) NOT NULL,
 7    first_name  VARCHAR(100) NOT NULL,
 8    last_name   VARCHAR(100) NOT NULL,
 9    created_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP
10);
11
12CREATE TABLE products (
13    id          SERIAL PRIMARY KEY,
14    name        VARCHAR(255) NOT NULL,
15    description TEXT NOT NULL,
16    price       DECIMAL(10,2) NOT NULL CHECK (price >= 0),
17    category    VARCHAR(50) NOT NULL,
18    stock       INTEGER NOT NULL DEFAULT 0
19);
20
21CREATE TABLE orders (
22    id           SERIAL PRIMARY KEY,
23    user_id      INTEGER REFERENCES users(id),
24    total_amount DECIMAL(10,2) NOT NULL,
25    status       VARCHAR(20) DEFAULT 'pending'
26);
27
28CREATE TABLE order_items (
29    id         SERIAL PRIMARY KEY,
30    order_id   INTEGER REFERENCES orders(id),
31    product_id INTEGER REFERENCES products(id),
32    quantity   INTEGER NOT NULL CHECK (quantity > 0),
33    price      DECIMAL(10,2) NOT NULL
34);
35
36-- ⚠️  Reviews table — schema exists, but API routes are empty
37CREATE TABLE reviews (
38    id         SERIAL PRIMARY KEY,
39    product_id INTEGER REFERENCES products(id),
40    user_id    INTEGER REFERENCES users(id),
41    rating     INTEGER NOT NULL CHECK (rating >= 1 AND rating <= 5),
42    comment    TEXT,
43    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
44);
