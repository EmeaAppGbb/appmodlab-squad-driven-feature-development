 1import Fastify from 'fastify';
 2import cors from '@fastify/cors';
 3import jwt from '@fastify/jwt';
 4import { config } from './config';
 5import { databasePlugin } from './plugins/database';
 6import { redisPlugin } from './plugins/redis';
 7import { productRoutes } from './modules/products/product.routes';
 8import { userRoutes } from './modules/users/user.routes';
 9import { orderRoutes } from './modules/orders/order.routes';
10
11const fastify = Fastify({ logger: { level: config.logLevel } });
12
13async function start() {
14  await fastify.register(cors, { origin: true });
15  await fastify.register(jwt, { secret: config.jwtSecret });
16  await fastify.register(databasePlugin);
17  await fastify.register(redisPlugin);
18
19  fastify.get('/health', async () => ({
20    status: 'ok', timestamp: new Date().toISOString()
21  }));
22
23  await fastify.register(productRoutes, { prefix: '/api/products' });
24  await fastify.register(userRoutes,    { prefix: '/api/users' });
25  await fastify.register(orderRoutes,   { prefix: '/api/orders' });
26  // ⚠️  reviewRoutes NOT registered yet — SQUAD will add this
27
28  await fastify.listen({ port: config.port, host: '0.0.0.0' });
29}
30
31start();
